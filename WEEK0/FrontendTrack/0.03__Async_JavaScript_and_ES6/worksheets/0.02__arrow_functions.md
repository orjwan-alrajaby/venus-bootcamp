Practice:

* converting functions to arrow functions

* Do NOT change the logic
* Use implicit return when possible

## 🧩 Challenge 6 — Promises & Async Code

### a)

Convert `.then()` and `.catch()` functions:

```js
const url = 'https://dummyjson.com/products';

function fetchData() {
  fetch(url)
    .then(function(response) {
      return response.json();
    })
    .then(function(data) {
      console.log(data);
    })
    .catch(function(error) {
      console.log(error);
    });
}
```

---

### b)

```js
function getUsers() {
  return fetch('https://dummyjson.com/users')
    .then(function(res) {
      return res.json();
    });
}
```

---

### c)

```js
fetch('https://dummyjson.com/posts')
  .then(function(res) {
    return res.json();
  })
  .then(function(posts) {
    console.log(posts.length);
  });
```
